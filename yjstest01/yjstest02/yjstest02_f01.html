FORM GET_DATA .

  SELECT CARRID, CONNID, COUNTRYFR, CITYFROM, AIRPFROM, COUNTRYTO, CITYTO,
         AIRPTO, FLTIME, DEPTIME,   ARRTIME,  DISTANCE, DISTID,    FLTYPE, PERIOD
    FROM ZSPFLI
   WHERE CARRID IN @S_CAR
     AND CONNID IN @S_CON
    INTO CORRESPONDING FIELDS OF TABLE @GT_DISP.

  SORT GT_DISP BY CARRID CONNID.


ENDFORM.

FORM SAVE_DATA .

  DATA : LV_CHECK.

  GO_ALV->CHECK_CHANGED_DATA(
    IMPORTING
      E_VALID   = LV_CHECK   " ALV ####  ##### 'X' ## ##### ###? " ### ### 'X' #####
  ).

  IF LV_CHECK IS INITIAL.
    MESSAGE 'Fail Save' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  LOOP AT GT_DELE INTO DATA(LS_DELE).
    DELETE FROM ZSPFLI WHERE CARRID = LS_DELE-CARRID
                         AND CONNID = LS_DELE-CONNID.
  ENDLOOP.

  LOOP AT GT_DISP INTO DATA(LS_DISP) WHERE CHANGED_DATA = ABAP_ON.

    GT_SAVE = VALUE #( BASE GT_SAVE
                ( CORRESPONDING #( LS_DISP ) )
              ).

  ENDLOOP.

  MODIFY ZSPFLI FROM TABLE GT_SAVE.

  IF SY-SUBRC EQ 0.
    COMMIT WORK AND WAIT.
    MESSAGE 'Success' TYPE 'S'.
    GO_ALV->REFRESH_TABLE_DISPLAY( ).

  ELSE.
    ROLLBACK WORK.
    MESSAGE 'Fail' TYPE 'S' DISPLAY LIKE 'E'.
    PERFORM GET_DATA.
    GO_ALV->REFRESH_TABLE_DISPLAY( ).

  ENDIF.


ENDFORM.

FORM DATA_CHANGED USING PO_DATA_CHANGED TYPE REF TO CL_ALV_CHANGED_DATA_PROTOCOL.

  DATA : LV_ERROR  TYPE C.


  " ### ### key ### ##
  LOOP AT PO_DATA_CHANGED->MT_INSERTED_ROWS INTO DATA(LS_INSERT).

    READ TABLE PO_DATA_CHANGED->MT_MOD_CELLS INTO DATA(LS_CAR) WITH KEY FIELDNAME = 'CARRID'
                                                                        ROW_ID    = LS_INSERT-ROW_ID.
    IF SY-SUBRC EQ 0.
      DATA(LV_CAR) = LS_CAR-VALUE.
    ELSE.
      PO_DATA_CHANGED->ADD_PROTOCOL_ENTRY(
        EXPORTING
          I_MSGID     = 'ZMSG_JS'
          I_MSGTY     = 'E'
          I_MSGNO     = '000'
          I_FIELDNAME = LS_CAR-FIELDNAME
          I_ROW_ID    = LS_INSERT-ROW_ID
      ).
      LV_ERROR = ABAP_ON.
    ENDIF.

    READ TABLE PO_DATA_CHANGED->MT_MOD_CELLS INTO DATA(LS_CON) WITH KEY FIELDNAME = 'CONNID'
                                                                        ROW_ID    = LS_INSERT-ROW_ID.

    IF SY-SUBRC EQ 0.
      DATA(LV_CON) = LS_CON-VALUE.
    ELSE.
      PO_DATA_CHANGED->ADD_PROTOCOL_ENTRY(
        EXPORTING
          I_MSGID     = 'ZMSG_JS'
          I_MSGTY     = 'E'
          I_MSGNO     = '000'
          I_FIELDNAME = LS_CON-FIELDNAME
          I_ROW_ID    = LS_INSERT-ROW_ID
      ).
      LV_ERROR = ABAP_ON.
    ENDIF.

    READ TABLE GT_DISP TRANSPORTING NO FIELDS WITH KEY CARRID = LV_CAR
                                                       CONNID = LV_CON.

    IF SY-SUBRC EQ 0.
      PO_DATA_CHANGED->ADD_PROTOCOL_ENTRY(
        EXPORTING
          I_MSGID     = 'ZMSG_JS'
          I_MSGTY     = 'E'
          I_MSGNO     = '000'
          I_FIELDNAME = LS_CON-FIELDNAME
          I_ROW_ID    = LS_INSERT-ROW_ID
      ).
      LV_ERROR = ABAP_ON.

    ELSE.

      PO_DATA_CHANGED->MODIFY_CELL(
        EXPORTING
          I_ROW_ID    = LS_INSERT-ROW_ID
          I_FIELDNAME = 'CHANGED_DATA'
          I_VALUE     = ABAP_ON
      ).

    ENDIF.

  ENDLOOP.

  " ## ### ### ##
  LOOP AT PO_DATA_CHANGED->MT_GOOD_CELLS INTO DATA(LS_MOD).

    CASE LS_MOD-FIELDNAME.

      WHEN 'CONNID'.

        IF LS_MOD-VALUE IS NOT INITIAL.

          READ TABLE PO_DATA_CHANGED->MT_GOOD_CELLS INTO DATA(LS_KEY) WITH KEY FIELDNAME = 'CARRID'
                                                                               ROW_ID    = LS_MOD-ROW_ID.
          IF SY-SUBRC EQ 0.

            READ TABLE GT_DISP WITH KEY CARRID = LS_KEY-VALUE
                                        CONNID = LS_MOD-VALUE
                                        TRANSPORTING NO FIELDS.
            IF SY-SUBRC EQ 0.
              PO_DATA_CHANGED->ADD_PROTOCOL_ENTRY(
                EXPORTING
                  I_MSGID     = 'ZMSG_JS'
                  I_MSGTY     = 'E'
                  I_MSGNO     = '001'
                  I_FIELDNAME = LS_MOD-FIELDNAME
                  I_ROW_ID    = LS_MOD-ROW_ID
              ).
              LV_ERROR = ABAP_ON.
            ENDIF.

          ENDIF.

        ENDIF.

    ENDCASE.

    PO_DATA_CHANGED->MODIFY_CELL(
      EXPORTING
        I_ROW_ID    = LS_MOD-ROW_ID
        I_FIELDNAME = 'CHANGED_DATA'
        I_VALUE     = ABAP_ON
    ).

  ENDLOOP.


  IF LV_ERROR EQ ABAP_ON.
    PO_DATA_CHANGED->DISPLAY_PROTOCOL( ).
  ENDIF.

  " ## ### ##
  LOOP AT PO_DATA_CHANGED->MT_DELETED_ROWS INTO DATA(LS_DEL).

    READ TABLE GT_DISP INTO DATA(LS_DEL_DISP) INDEX LS_DEL-ROW_ID.
    GT_DELE = VALUE #( BASE GT_DELE
                ( CORRESPONDING #( LS_DEL_DISP ) )
              ).

  ENDLOOP.



*--------------------------------------------------------------------*
* PBO ### ### ##
* ## ## Error log # ##
*--------------------------------------------------------------------*
*  LEAVE SCREEN.

ENDFORM.
