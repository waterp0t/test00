FORM GET_DATA .

  SELECT CARRID, CONNID, COUNTRYFR, CITYFROM, AIRPFROM, COUNTRYTO, CITYTO,
         AIRPTO, FLTIME, DEPTIME,   ARRTIME,  DISTANCE, DISTID,    FLTYPE, PERIOD
    FROM ZSPFLI
   WHERE CARRID IN @S_CAR
     AND CONNID IN @S_CON
    INTO CORRESPONDING FIELDS OF TABLE @GT_DISP.

  SORT GT_DISP BY CARRID CONNID.


  IF GT_SGEO[] IS INITIAL.

    SELECT CITY COUNTRY
      FROM SGEOCITY
      INTO CORRESPONDING FIELDS OF TABLE GT_SGEO.

    DELETE ADJACENT DUPLICATES FROM GT_SGEO COMPARING CITY COUNTRY.

  ENDIF.

  IF GT_SAIR[] IS INITIAL.

    SELECT ID NAME
      FROM SAIRPORT
      INTO CORRESPONDING FIELDS OF TABLE GT_SAIR.

    DELETE ADJACENT DUPLICATES FROM GT_SAIR COMPARING ID NAME.

  ENDIF.

ENDFORM.

FORM SAVE_DATA .

  DATA : LV_CHECK,
         LV_REP.

  GO_ALV->CHECK_CHANGED_DATA(
    IMPORTING
      E_VALID   = LV_CHECK   " ALV ####  ##### 'X' ## ##### ###? " ### ### 'X' #####
  ).

  IF LV_CHECK IS INITIAL.
    MESSAGE 'Fail Save' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ELSEIF GV_CHK IS INITIAL.
    EXIT.
  ENDIF.

  LOOP AT GT_DELE INTO DATA(LS_DELE).
    DELETE FROM ZSPFLI WHERE CARRID = LS_DELE-CARRID
                         AND CONNID = LS_DELE-CONNID.
  ENDLOOP.

  LOOP AT GT_DISP INTO DATA(LS_DISP) WHERE CHANGED_DATA = ABAP_ON.

    GT_SAVE = VALUE #( BASE GT_SAVE
                ( CORRESPONDING #( LS_DISP ) )
              ).

  ENDLOOP.

  MODIFY ZSPFLI FROM TABLE GT_SAVE.

  IF SY-SUBRC EQ 0.
    COMMIT WORK AND WAIT.
    MESSAGE 'Success' TYPE 'S'.
    GO_ALV->REFRESH_TABLE_DISPLAY( ).

    IMPORT E_CHK TO LV_REP FROM MEMORY ID 'CHK'.
    IF LV_REP EQ ABAP_ON.
      EXPORT E_CHK FROM 'S' TO MEMORY ID 'CHK'.
    ENDIF.

  ELSE.
    ROLLBACK WORK.
    MESSAGE 'Fail' TYPE 'S' DISPLAY LIKE 'E'.
    PERFORM GET_DATA.
    GO_ALV->REFRESH_TABLE_DISPLAY( ).

  ENDIF.

  CLEAR GV_CHK.

ENDFORM.
